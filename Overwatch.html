import pandas as pd
import types
import numpy as np
import matplotlib.pyplot as plt

def wins_by(df, *x):
    return df.dropna().groupby(list(x)).Result.agg([
        ('Win rate', lambda x:x.mean()*100),
        ('Total', 'count'),
        ('Wins', lambda x:x.sum()),
        ('Losses', lambda x:len(x)-x.sum()),
        ('Delta', lambda x:x.sum()*2-len(x)),
    ])
pd.DataFrame.wins_by = wins_by

cb = lambda series:(series > 0).map({True: 'blue', False: 'red'})
